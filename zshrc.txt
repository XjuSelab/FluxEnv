# ------ 2.2 C++ 编译助手 & 快捷键 ------
# 找到最新修改的 cpp 文件
last_cpp() { ls -t *.cpp 2>/dev/null | head -n 1; }

# F10: 只编译
c() {
    local file=$(last_cpp)
    [ -n "$file" ] && { echo "Compiling $file ..."; g++ "$file" -o a.out && echo "Done."; } || echo "No .cpp file found."
}

# Shift+F10: 编译并运行
cr() {
    local file=$(last_cpp)
    [ -n "$file" ] && { echo "Compiling $file ..."; g++ "$file" -o a.out && { echo "Running ./a.out ..."; ./a.out; }; } || echo "No .cpp file found."
}

# F9: 查看代码
cc() {
    local file=$(last_cpp)
    [ -n "$file" ] && { echo "===== $file ====="; cat "$file"; } || echo "No .cpp file found."
}

# 自动补全定义
compdef -d c
compdef -d cr
compdef -d cc

# 快捷键绑定
bindkey -s '^[[21~' 'c\n'           # F10

bindkey -s '^[[20~' 'cc\n'          # F9
bindkey -s '^[[24~' 'build-compile\n' # F12

# ------ 2.3 数据库 (PostgreSQL / Kingbase) ------
initpsql() {
    echo "Check PostgreSQL cluster status..."; pg_lsclusters
    echo "Start PostgreSQL 14 main..."; sudo pg_ctlcluster 14 main start
    echo "Welcome to PostgreSQL "; pg_lsclusters
}

enterpsql() {
    echo "Connecting to PostgreSQL using pgcli..."
    pgcli -h localhost -p 5433 -U postgres -d postgres
}

install_kingbase_types() {
    echo "🚀 Installing kingbase_types.sql into all databases..."
    for db in $(psql -h localhost -p 5433 -U postgres -t -c "SELECT datname FROM pg_database WHERE datistemplate = false;"); do
        echo "👉 Installing into $db"
        psql -h localhost -p 5433 -U postgres -d "$db" -f ~/kingbase_types.sql
    done
    echo "✅ Done."
}
